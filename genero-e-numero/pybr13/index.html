def le_csv(arquivo):
    with open(arquivo, encoding='utf-8') as fobj:
        return [linha.split(',')
                for linha in fobj.readlines()]

def escreve_csv(linhas):
    with open(arquivo, mode='w', encoding='utf-8') as fobj:
        for linha in linhas:
            fobj.writeline(','.join(linha))
